PROCESS
function auditSession(session):

  checkpoints = getCheckpoints(session.phase)

  passed = 0
  total = checkpoints.length

  for checkpoint in checkpoints:
      if validate(checkpoint, session):
          passed += 1
      else:
          if checkpoint.isCritical:
              return {
                status: "FAIL",
                complianceScore: 0,
                reason: checkpoint.name
              }

  score = (passed / total) * 100

  if score == 100:
      return PASS
  else:
      return FAIL
